"""
interpretador_ia.py — Interpretador de projetos elétricos RGE/CPFL via Claude API.
Espelha a lógica do OpenAICroquiInterpreter.php usando o Anthropic Python SDK.
"""
import os, json, base64, re
from typing import Optional

    import fitz  # PyMuPDF
    _FITZ_OK = True
except ImportError:
    _FITZ_OK = False

import anthropic

# ─────────────────────────────────────────────
#  EXTRAÇÃO DO PDF
# ─────────────────────────────────────────────

def _extrair_pdf_bundle(caminho_pdf: str) -> dict:
    if not _FITZ_OK:
        raise RuntimeError("PyMuPDF não instalado. Execute: pip install pymupdf")
    doc = fitz.open(caminho_pdf)
    texto = ""
    for page in doc:
        texto += page.get_text()
    texto = texto[:90000]

    imagens = []
    for i, page in enumerate(doc):
        if i >= 4:
            break
        pix = page.get_pixmap(dpi=120)
        b64 = base64.standard_b64encode(pix.tobytes("png")).decode()
        imagens.append(f"data:image/png;base64,{b64}")

    doc.close()
    return {"text": texto, "images": imagens}


# ─────────────────────────────────────────────
#  API CLAUDE
# ─────────────────────────────────────────────

def interpretar_pdf(caminho_pdf: str, api_key: Optional[str] = None, progresso=None) -> dict:
    """Interpreta um PDF de projeto RGE/CPFL e retorna JSON estruturado."""
    if progresso:
        progresso("Extraindo texto e imagens do PDF...")
    bundle = _extrair_pdf_bundle(caminho_pdf)
    return interpretar_texto(bundle["text"], bundle["images"], api_key=api_key, progresso=progresso)


def interpretar_texto(texto_pdf: str, imagens: list = None, api_key: Optional[str] = None, progresso=None) -> dict:
    """Envia texto+imagens para o Claude e retorna o JSON do projeto."""
    api_key = api_key or os.environ.get("ANTHROPIC_API_KEY", "")
    if not api_key:
        raise RuntimeError(
            "Configure ANTHROPIC_API_KEY no ambiente ou passe api_key= para usar a interpretação por IA."
        )

    client = anthropic.Anthropic(api_key=api_key)
    texto_pdf = texto_pdf[:90000]

    content = [
        {
            "type": "text",
            "text": (
                "Analise este projeto elétrico RGE/CPFL e gere o JSON estruturado para croqui automático.\n\n"
                "TEXTO EXTRAÍDO DO PDF:\n" + texto_pdf
            )
        }
    ]

    for img_url in (imagens or [])[:4]:
        if img_url.startswith("data:"):
            media_type = img_url.split(";")[0].split(":")[1]
            b64_data = img_url.split(",", 1)[1]
            content.append({
                "type": "image",
                "source": {"type": "base64", "media_type": media_type, "data": b64_data},
            })

    content.append({
        "type": "text",
        "text": (
            "\n\nRetorne SOMENTE JSON válido, sem markdown, sem explicações. "
            "Estrutura obrigatória: {meta, nos, trechos, equipamentos, areas, textos}"
        )
    })

    if progresso:
        progresso("Enviando para IA Claude (pode levar 1-3 min)...")

    with client.messages.stream(
        model="claude-opus-4-8",
        max_tokens=16000,
        thinking={"type": "adaptive"},
        system=_system_prompt(),
        messages=[{"role": "user", "content": content}],
    ) as stream:
        message = stream.get_final_message()

    json_text = ""
    for block in message.content:
        if block.type == "text":
            json_text += block.text

    json_text = json_text.strip()
    # Limpar blocos markdown se presentes
    json_text = re.sub(r'^```\w*\n?', '', json_text)
    json_text = re.sub(r'\n?```$', '', json_text).strip()

    obj = json.loads(json_text)
    if not isinstance(obj, dict):
        raise RuntimeError(f"IA respondeu mas JSON inválido: {json_text[:400]}")

    return sanitizar_projeto(obj)


# ─────────────────────────────────────────────
#  SANITIZAÇÃO (porta do sanitizeProject PHP)
# ─────────────────────────────────────────────

def sanitizar_projeto(obj: dict) -> dict:
    # Garante tipos corretos independente do que a IA retornou
    for key in ["meta", "nos", "trechos", "equipamentos", "areas", "textos"]:
        esperado_dict = (key == "meta")
        val = obj.get(key)
        if esperado_dict:
            if not isinstance(val, dict):
                obj[key] = {}
        else:
            if not isinstance(val, list):
                obj[key] = []

    obj["nos"] = _dedupe_by_id(obj["nos"])
    obj["nos"] = _normalize_nodes(obj["nos"])
    obj["trechos"] = _normalize_trechos(obj["trechos"])
    obj["equipamentos"] = _normalize_equipamentos(obj["equipamentos"])
    obj["areas"] = _normalize_areas(obj["areas"])

    obj["nos"] = _dedupe_by_id(obj["nos"])
    node_set = {n["id"] for n in obj["nos"]}

    # Criar nós faltantes referenciados em trechos
    for t in obj["trechos"]:
        for k in ("de", "para"):
            nid = str(t.get(k, ""))
            if nid and re.match(r'^P\d+$', nid) and nid not in node_set:
                obj["nos"].append({
                    "id": nid, "tipo": "POSTE_EXISTENTE",
                    "x": "", "y": "", "label": nid, "observacao": "Auto-criado"
                })
                node_set.add(nid)

    # Fallback para equipamentos sem nó válido
    fallback = next((n["id"] for n in obj["nos"] if re.match(r'^P\d+$', n["id"])), "")
    for eq in obj["equipamentos"]:
        if not eq.get("no_id") or eq["no_id"] not in node_set:
            eq["no_id"] = fallback

    # Remover trechos inválidos
    obj["trechos"] = [
        t for t in obj["trechos"]
        if t.get("de") and t.get("para")
        and t["de"] in node_set and t["para"] in node_set
        and t["de"] != t["para"]
    ]

    return obj


def _extract_p(text: str) -> str:
    m = re.search(r'\bP\s*0*(\d+)\b', str(text), re.I)
    return f"P{int(m.group(1))}" if m else ""


def _normalize_nodes(rows: list) -> list:
    out = []
    for row in rows:
        raw = str(row.get("id") or row.get("label") or "")
        pid = _extract_p(raw)
        if not pid:
            continue
        row["id"] = pid
        if not row.get("label"):
            row["label"] = pid
        if not row.get("tipo"):
            row["tipo"] = "POSTE_EXISTENTE"
        out.append(row)
    return out


def _normalize_trechos(rows: list) -> list:
    out = []
    for row in rows:
        de = _extract_p(str(row.get("de", "")))
        para = _extract_p(str(row.get("para", "")))
        if (not de or not para):
            texto = " ".join(str(v) for v in row.values())
            m = re.search(r'V\s*(\d+)\s*[-–]\s*(\d+)', texto, re.I)
            if m:
                de = f"P{int(m.group(1))}"
                para = f"P{int(m.group(2))}"
        if not de or not para:
            continue
        row["de"] = de
        row["para"] = para
        if not row.get("tipo"):
            row["tipo"] = "MT"
        out.append(row)
    return out


def _normalize_equipamentos(rows: list) -> list:
    for row in rows:
        p = _extract_p(str(row.get("no_id", "")))
        if p:
            row["no_id"] = p
    return rows


def _normalize_areas(rows: list) -> list:
    for row in rows:
        ids = []
        for part in re.split(r'[|,;\s]+', str(row.get("nos", ""))):
            p = _extract_p(part)
            if p:
                ids.append(p)
        row["nos"] = "|".join(dict.fromkeys(ids))
    return rows


def _dedupe_by_id(rows: list) -> list:
    seen = set()
    out = []
    for row in rows:
        pid = str(row.get("id", "")).strip()
        if not pid or pid in seen:
            continue
        seen.add(pid)
        out.append(row)
    return out
